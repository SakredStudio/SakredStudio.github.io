// ─── HELPERS ─────────────────────────────────────────────────────────────────
const getIdol = (id: string|null) => IDOLS.find(i => i.id === id);

const getDays = (dateStr: string) => {
  const [y,m,d] = dateStr.split('-').map(Number);
  const target = new Date(y, m-1, d);
  const today = new Date(); today.setHours(0,0,0,0);
  return Math.ceil((target.getTime()-today.getTime())/86400000);
};

const openURL = (url: string) => Linking.openURL(url).catch(() => Alert.alert('Could not open link'));

// ─── GLOBAL STORE (AsyncStorage-backed, no extra lib) ──────────────────────
const STORE_KEY = 'fandrop_v1';
const defaultState: AppStore = {
  myIdols:['bts','bp'], savedEvent:null,
  checkedItems:{}, wishlist:[], hasOnboarded:false,
};
let _state = {...defaultState};
let _listeners: Array<()=>void> = [];

const store = {
  get: () => _state,
  set: (update: Partial<AppStore>) => {
    _state = {..._state, ...update};
    _listeners.forEach(l=>l());
    AsyncStorage.setItem(STORE_KEY, JSON.stringify(_state)).catch(()=>{});
  },
  sub: (fn: ()=>void) => { _listeners.push(fn); return ()=>{_listeners=_listeners.filter(l=>l!==fn)}; },
};

AsyncStorage.getItem(STORE_KEY).then(s=>{
  if(s){ _state={...defaultState,...JSON.parse(s)}; _listeners.forEach(l=>l()); }
}).catch(()=>{});

function useStore<T>(sel: (s:AppStore)=>T): T {
  const [v,setV] = useState(()=>sel(store.get()));
  useEffect(()=>store.sub(()=>setV(sel(store.get()))), []);
  return v;
}

// ─── API ─────────────────────────────────────────────────────────────────────
// ⚠️ For production: replace this URL with your Cloudflare Worker proxy
// NEVER ship with the API key in the binary for a public release
const CLAUDE_URL = 'https://api.anthropic.com/v1/messages';
const API_KEY = process.env.EXPO_PUBLIC_ANTHROPIC_API_KEY || '';

async function callClaude(system:string, messages:ChatMsg[], maxTokens=1000): Promise<string> {
  const res = await fetch(CLAUDE_URL, {
    method:'POST',
    headers:{
      'Content-Type':'application/json',
      'x-api-key': API_KEY,
      'anthropic-version':'2023-06-01',
    },
    body:JSON.stringify({model:'claude-sonnet-4-20250514', max_tokens:maxTokens, system, messages}),
  });
  if(!res.ok) throw new Error(`HTTP ${res.status}`);
  const data = await res.json();
  return data.content?.find((b:any)=>b.type==='text')?.text || '';
}

// ─── ROOT ─────────────────────────────────────────────────────────────────────
export default function App() {
  const [fontsLoaded] = useFonts({
    Syne_700Bold, Syne_800ExtraBold, Syne_900Black,
    DMSans_400Regular, DMSans_500Medium, DMSans_600SemiBold,
    SpaceMono_400Regular, SpaceMono_700Bold,
  });
  const [appReady, setAppReady] = useState(false);
  const hasOnboarded = useStore(s=>s.hasOnboarded);
  const myIdols = useStore(s=>s.myIdols);
  const primaryColor = (myIdols[0] ? getIdol(myIdols[0])?.color : null) || '#7c3aed';

  useEffect(()=>{
    if(fontsLoaded){
      SplashScreen.hideAsync();
      setAppReady(true);
    }
  }, [fontsLoaded]);

  if(!appReady) return null;

  if(!hasOnboarded) return (
    <SafeAreaProvider>
      <StatusBar style="light"/>
      <OnboardingScreen/>
    </SafeAreaProvider>
  );

  return (
    <SafeAreaProvider>
      <StatusBar style="light"/>
      <NavigationContainer theme={NAV_THEME}>
        <TabNavigator primaryColor={primaryColor}/>
      </NavigationContainer>
    </SafeAreaProvider>
  );
}
